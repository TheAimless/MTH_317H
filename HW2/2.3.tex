\begin{question}
    \normalfont

    Assume that $W_1$ and $W_2$ are both subspaces of a vector space, $V$. Prove that $U=W_1\intersect W_2$ is also a subspace of $V$.
\end{question}

\renewcommand\qedsymbol{$\blacksquare$}

    \begin{enumerate}

    \end{enumerate}
\end{proof}
